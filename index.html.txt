<script>
function calcular() {
  let total =
    (val("b50") * 50) +
    (val("b20") * 20) +
    (val("b10") * 10) +
    (val("b5") * 5) +
    (val("m1") * 1) +
    (val("m50") * 0.5) +
    (val("m20") * 0.2) +
    (val("m10") * 0.1) +
    (val("m5") * 0.05);

  document.getElementById("total").innerText = total.toFixed(2);
}

function val(id) {
  return parseFloat(document.getElementById(id).value) || 0;
}

// Escuchar cambios
document.querySelectorAll("input").forEach(input => {
  input.addEventListener("input", calcular);
});

// Inicial
calcular();
